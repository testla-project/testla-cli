## CORRECT task file pattern

import { Actor, Task } from '@testla/screenplay-playwright';
import { Navigate, Click, Fill } from '@testla/screenplay-playwright/web';
import { LoginScreen } from '../screens/login-screen';

export class LoginTask extends Task {
    private constructor() { super(); }

    public async performAs(actor: Actor): Promise<any> {
        return actor.attemptsTo(
            Navigate.to(process.env.BASE_URL ?? 'https://example.com'),
            Fill.in(LoginScreen.USERNAME, 'tomsmith'),
            Fill.in(LoginScreen.PASSWORD, 'SuperSecretPassword!'),
            Click.on(LoginScreen.LOGIN_BUTTON),
        );
    }

    public static toApp(): LoginTask { return new LoginTask(); }
}

## Rules
- Fill syntax: Fill.in(Screen.PROP, 'value') — NEVER Fill.in(...).with(...)
- performAs MUST use: return actor.attemptsTo(...)
- Bob comes from the fixture: async ({ Bob }) => {}
- NEVER use: new testla.screenplay.Actor(), bob.usesAbilities(), describe(), it()
- Element.toBe.visible() argument MUST be a Screen.PROP — read the Screen file to find the right one
